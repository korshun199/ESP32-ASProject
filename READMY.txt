from fastapi import FastAPI, WebSocket, WebSocketDisconnect
import time
import asyncio

app = FastAPI()

# Хранилище активного подключения
class ConnectionManager:
    def __init__(self):
        self.active_connection: WebSocket = None

    async def connect(self, websocket: WebSocket):
        await websocket.accept()
        self.active_connection = websocket

    def disconnect(self):
        self.active_connection = None

    async def send_command(self, cmd_id: int, param: int):
        if self.active_connection:
            # TTL: шлем текущее время сервера
            data = {
                "cmd": cmd_id,
                "val": param,
                "time": int(time.time())
            }
            await self.active_connection.send_json(data)
            return True
        return False

manager = ConnectionManager()

@app.websocket("/ws")
async def websocket_endpoint(websocket: WebSocket):
    await manager.connect(websocket)
    print("ESP32 подключилась!")
    try:
        while True:
            # Ждем данных или просто держим связь
            data = await websocket.receive_text()
            print(f"Ответ от ESP32: {data}")
    except WebSocketDisconnect:
        manager.disconnect()
        print("ESP32 отключилась")

# Эндпоинт для тебя (вызывать через браузер или curl)
# Пример: http://твой-vps/send?id=1&val=100
@app.get("/send")
async def trigger_command(id: int, val: int):
    success = await manager.send_command(id, val)
    return {"status": "sent" if success else "esp_offline"}

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8000)

2. Сервер на Python (VPS)
Используем стандартный модуль socket. Он позволит нам видеть состояние соединения на уровне ОС.

Python
import socket
import time

HOST = '0.0.0.0'  # Слушать все интерфейсы
PORT = 8000       # Порт для ESP32

def start_server():
    with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as s:
        s.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
        s.bind((HOST, PORT))
        s.listen()
        print(f"Сервер запущен на порту {PORT}. Ожидание ESP32...")
        
        while True:
            conn, addr = s.accept()
            with conn:
                print(f"ESP32 подключилась: {addr}")
                conn.settimeout(10.0) # Тайм-аут на чтение (проверка связи)
                try:
                    while True:
                        # Здесь ты можешь вводить команду вручную для теста
                        # В реальности тут будет логика приема из веб-интерфейса
                        cmd = input("Введите команду (или 'wait'): ")
                        if cmd != 'wait':
                            # Отправляем как сырые байты. Напр: "1:255\n"
                            conn.sendall(cmd.encode() + b'\n')
                        
                        # Небольшая пауза, чтобы не забивать цикл
                        time.sleep(0.1)
                except (ConnectionResetError, BrokenPipeError, socket.timeout):
                    print("Связь с ESP32 потеряна.")
                    continue

if __name__ == "__main__":
    start_server()
3. Клиент на ESP32 (Низкий уровень)
Здесь мы используем WiFiClient, который работает напрямую с TCP.

C++
#include <WiFi.h>

const char* ssid = "YOUR_SSID";
const char* password = "YOUR_PASSWORD";
const char* host = "YOUR_VPS_IP";
const uint16_t port = 8000;

WiFiClient client;
bool wasConnected = false;

void setup() {
  Serial.begin(115200);
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) { delay(500); }
  Serial.println("Wi-Fi OK");
}

void loop() {
  if (!client.connected()) {
    if (wasConnected) {
      Serial.println("Lost Connection");
      wasConnected = false;
    }
    
    // Попытка переподключения
    if (client.connect(host, port)) {
      Serial.println("Connected to VPS");
      wasConnected = true;
    } else {
      delay(2000); // Пауза перед повтором
      return;
    }
  }

  // Читаем данные из сокета
  while (client.available()) {
    String line = client.readStringUntil('\n');
    Serial.print("Получено: ");
    Serial.println(line);
    
    // Квитирование: отвечаем серверу, что получили (ACK)
    client.print("ACK\n");
  }
}
Почему это лучше для твоей задачи:
Минимум памяти: Мы не парсим JSON, не держим заголовки HTTP. Просто байты в трубе.

Мгновенная реакция: Как только сервер сделал sendall, данные уже в буфере ESP32.

Честный статус: Если кабель на роутере выдернут, TCP-стек через время (или по отсутствию ACK) поймет это, и client.connected() станет false.

Олежка, теперь важный момент про TTL (время жизни):
В низкоуровневых командах принято добавлять первым байтом "номер команды" или "время". Например, команда 1712656400:1:255. ESP32 берет первые 10 цифр, сравнивает со своим временем (если синхронизирована по NTP) и решает, не "протухла" ли команда.

import socket
import threading
from flask import Flask, request

app = Flask(__name__)
command_buffer = None  # Наш буфер для одной команды

# TCP Сервер для ESP32 в отдельном потоке
def tcp_server():
    global command_buffer
    with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as s:
        s.bind(('0.0.0.0', 8000))
        s.listen()
        while True:
            conn, addr = s.accept()
            with conn:
                print(f"ESP32 Connected: {addr}")
                while True:
                    if command_buffer:
                        conn.sendall(command_buffer.encode() + b'\n')
                        command_buffer = None # Очищаем буфер после отправки
                    try:
                        conn.settimeout(0.1)
                        conn.recv(1024) # Проверка связи
                    except socket.timeout:
                        continue
                    except: break

# API для телефона
@app.route('/set')
def set_command():
    global command_buffer
    cmd = request.args.get('c')
    val = request.args.get('v')
    command_buffer = f"{cmd}:{val}"
    return f"Команда {command_buffer} добавлена в очередь!"

if __name__ == "__main__":
    # Запускаем TCP сервер в фоне
    threading.Thread(target=tcp_server, daemon=True).start()
    # Запускаем Web-интерфейс для Андроида

import socket
import threading
from flask import Flask, request

app = Flask(__name__)
# Тут храним последнюю команду, пока ESP её не заберет
cmd_queue = [] 

# --- TCP СЕРВЕР ДЛЯ ESP32 ---
def esp32_tcp_handler():
    global cmd_queue
    # Создаем сокет (IPv4, TCP)
    with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as s:
        s.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
        s.bind(('0.0.0.0', 8000)) # Порт для ESP32
        s.listen(1)
        
        while True:
            print("Ожидание подключения ESP32...")
            conn, addr = s.accept()
            with conn:
                print(f"ESP32 подключена: {addr}")
                conn.setblocking(False) # Не ждем вечно
                
                while True:
                    # Если в очереди есть команды - отправляем их все
                    while cmd_queue:
                        command = cmd_queue.pop(0)
                        try:
                            conn.sendall(command.encode() + b'\n')
                            print(f"Отправлено на ESP32: {command}")
                        except:
                            print("Ошибка отправки, ESP32 отвалилась")
                            break
                    
                    # Проверяем, жива ли еще связь (короткий recv)
                    try:
                        data = conn.recv(1024)
                        if not data: break # Если пришел пустой пакет - дисконнект
                    except BlockingIOError:
                        pass # Данных нет, это нормально
                    except Exception:
                        break
                    
                    import time
                    time.sleep(0.1) # Чтобы не раскалять процессор

# --- WEB ИНТЕРФЕЙС ДЛЯ ANDROID ---
@app.route('/cmd')
def web_trigger():
    c = request.args.get('c', '0') # Номер команды
    v = request.args.get('v', '0') # Значение (параметр)
    
    formatted_cmd = f"{c}:{v}"
    cmd_queue.append(formatted_cmd)
    
    return f"<h3>Команда [{formatted_cmd}] в очереди!</h3><p>ESP32 заберет её через мгновение.</p>"

if __name__ == "__main__":
    # Запускаем TCP сервер в отдельном потоке
    threading.Thread(target=esp32_tcp_handler, daemon=True).start()
    # Запускаем Flask на порту 5000
    app.run(host='0.0.0.0', port=5000)

Давай разберемся, почему так происходит. Есть две вероятные причины:

1. Проблема "Новой строки" и Буферизации
В моем коде выше отправка происходит только тогда, когда в списке cmd_queue что-то появляется через Web-интерфейс (Flask).

Если ты просто запустил сервер и клиент, клиент будет висеть на s.recv(1024) вечно, пока ты не зайдешь в браузер и не перейдешь по ссылке:
http://твой_vps_ip:5000/cmd?c=hello&v=world

2. Маленький баг с блокировкой
Посмотри на эту строчку в серверном коде:
conn.setblocking(False)

В некоторых версиях Python и ОС, если переключить сокет в неблокирующий режим сразу после accept, conn.recv может выкидывать ошибку, которую мы ловим в except, и цикл может работать не так, как ожидалось.

Давай сделаем супер-надежный тест "Ловца":
Замени в сервере (файл server.py) блок esp32_tcp_handler на этот упрощенный вариант (без неблокирующего режима, для теста):

Python
def esp32_tcp_handler():
    global cmd_queue
    with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as s:
        s.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
        s.bind(('0.0.0.0', 8000))
        s.listen(1)
        while True:
            conn, addr = s.accept()
            with conn:
                print(f"ESP32 подключена: {addr}")
                while True:
                    if cmd_queue:
                        command = cmd_queue.pop(0)
                        conn.sendall(command.encode() + b'\n')
                        print(f"Отправлено: {command}")
                    
                    # Чтобы не вешать поток, просто ждем чуть-чуть
                    import time
                    time.sleep(0.5)
